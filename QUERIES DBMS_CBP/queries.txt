 CREATE TABLE MOVIE(
  MOVIE_ID VARCHAR(3) CONSTRAINT P1 PRIMARY KEY,
  TITLE VARCHAR(20) NOT NULL,
   RELEASE_DATE DATE NOT NULL,
   RATING VARCHAR(8) NOT NULL,
   DURATION VARCHAR(10)NOT NULL
   );
   CREATE TABLE ACTOR(
   ACTOR_ID VARCHAR(3) CONSTRAINT P2 PRIMARY KEY,
   NAME VARCHAR(30) NOT NULL,
   GENDER VARCHAR(1) NOT NULL,
   DOB DATE NOT NULL
   );
  
  CREATE TABLE CAST
   (
    MOVIE_ID VARCHAR(3),
    ACTOR_ID VARCHAR(3),
    ROLE VARCHAR(10) NOT NULL,
    CONSTRAINT F1 FOREIGN KEY(MOVIE_ID) REFERENCES MOVIE(MOVIE_ID),
    CONSTRAINT F2 FOREIGN KEY(ACTOR_ID) REFERENCES ACTOR(ACTOR_ID)
   );

create table producer
    (
      producer_id varchar(3) constraint p3 primary key,
      name varchar(20) not null,
      films number(3) not null
    );
 CREATE TABLE PRODUCTION(
    MOVIE_ID VARCHAR(3),
    PRODUCER_ID VARCHAR(3), 
    BUDGET VARCHAR(10) NOT NULL, 
    CONSTRAINT F3 FOREIGN KEY (MOVIE_ID) REFERENCES MOVIE(MOVIE_ID),
    CONSTRAINT F4 FOREIGN KEY(PRODUCER_ID) REFERENCES PRODUCER(PRODUCER_ID)
   );

 create table direction_crew
    (
      director_id varchar(3) constraint p4 primary key,
      name varchar(20) not null,
      gender varchar(1) not null
    );
 CREATE TABLE DIRECTION (
      MOVIE_ID VARCHAR(3), 
      DIRECTOR_ID VARCHAR(3),
      DIREC_ROLE VARCHAR(15),
      CONSTRAINT F5 FOREIGN KEY(MOVIE_ID) REFERENCES MOVIE(MOVIE_ID),
      CONSTRAINT F6 FOREIGN KEY (DIRECTOR_ID) REFERENCES DIRECTION_CREW(DIRECTOR_ID)
   );
